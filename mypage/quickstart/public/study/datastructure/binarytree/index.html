#include<stdio.h>
#include<stdlib.h>
typedef char AdjType;
typedef struct Btree{
	AdjType date;
	struct Btree *lchild,*rchild;
}btree;
int count;
int i;
btree *Init(){
	btree *p;
	char x;
	scanf("%c",&x);getchar();
	if(x ==' '){
		p=NULL;	
	}
	else{
		p=(btree *)malloc(sizeof(btree));
		p->date=x;
		printf("\n\t\t\t请输入一个非#的字符,当前访问到%c的左子女",p->date);
		p->lchild=Init();
		printf("\n\t\t\t请输入一个非#的字符,当前访问到%c的右子女",p->date);
		p->rchild=Init();
	}
	return p;
}
void xianxu(btree *p){
	if(p==NULL)
		return;
	printf("%c",p->date);
	xianxu(p->lchild);
	xianxu(p->rchild);
}
void zhongxu(btree *p){
	if(p==NULL)
		return;
	zhongxu(p->lchild);
	printf("%c",p->date);
	zhongxu(p->rchild);
}
void houxu(btree *p){
	if(p==NULL)
		return;
	houxu(p->lchild);
	houxu(p->rchild);
	printf("%c",p->date);
}
int yezi(btree *p){
	if(p!=NULL){
		if(p->lchild==NULL&&p->rchild==NULL){
			count++;
			return count;
		}
		else{
			yezi(p->lchild);
			yezi(p->rchild);
		}
	}
	else{
		return i;
	}
}
int bNodecount(btree *p){
	if(p!=NULL){
		count++;
		bNodecount(p->lchild);
		bNodecount(p->rchild);
		return count;
	}
}
int Treedeep(btree *p){
	int ldeep,rdeep;
	if(p==NULL)
		return 0;
	ldeep=Treedeep(p->lchild);
	rdeep=Treedeep(p->rchild);
	if(ldeep>rdeep)
		return (ldeep+1);
	else
		return (rdeep+1);
}
